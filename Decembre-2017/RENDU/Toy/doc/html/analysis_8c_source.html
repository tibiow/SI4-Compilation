78 static bool compatible_types_for_classes(ast_node *t1, ast_node *t2) {
79  // If type names are identical, t1 and t2 are compatible (ugly and not very
80  // efficient. We should use a same descriptor in this case). Must be fixed?
81  if (strcmp(TYPE_NAME(t1), TYPE_NAME(t2)) == 0) return true;
82 
83  // If t1 is exactly an Object, everybody is compatible to it (but needs a cast)
84  if (AST_TYPE(t1) == Object_class) return true;
85 
86  // if t2 is exactly an Object return false since t1 is not exactly an Object
87  if (AST_TYPE(t2) == Object_class) return false;
88 
89  // See if t2 has t1 in its inheritance list
90  ast_node *klass = symbol_table_search(TYPE_NAME(t2));
91  ast_node *parent = symbol_table_search(IDENT_VAL(CLASS_NAME(CLASS_PARENT(klass))));
92 
93  if (parent!=Object_class) return compatible_types_for_classes(t1, AST_TYPE(parent));
94 
95  return false;
96 }

99 static bool compatible_types(ast_node *e1, ast_node *e2) {
100  ast_node *t1 = AST_TYPE(e1), *t2 = AST_TYPE(e2);
101 
102  if (!valid_type(t1) || !valid_type(t2)) // to avoid error cascades => true
103  return true;
104 
105  if (t1 == t2) return true; // same types => true
106 
107  // Testing a mix of integer and float computations
108  if ((t1 == int_type && t2 == float_type) ||
109  (t1 == float_type && t2 == int_type)) return true;
110 
111  // Testing if t1 is an object and t2 is null
112  if (!TYPE_IS_STANDARD(t1) && (t2 == null_type)) return true;
113 
114  // Testing objects
115  if (! TYPE_IS_STANDARD(t1) && !TYPE_IS_STANDARD(t2)) { // both are objects
116  if (compatible_types_for_classes(t1, t2)) {
117  if (t1 != t2) AST_CAST(e2) = t1;
118  return true;
119  }
120  }
121 
122  // Since we could not return true, return false
123  return false;
124 }

141 static void prefix_analysis(ast_node *node, char *id){
142  // Type the prefix
143  analysis(IDENT_PREFIX(node));
144 
145  // Verify that the prefix is a class instance
146  ast_node *prefix_type = AST_TYPE(IDENT_PREFIX(node));
147 
148  AST_TYPE(node) = NULL; // will be changed if no error is detected
149 
150  if (!valid_type(prefix_type)) return;
151 
152  if (TYPE_IS_STANDARD(prefix_type)) {
153  error_msg(node, "prefix of '%s' is not a class instance", id);
154  return;
155  }
156 
157  // Verify that the class has the given identifier as field
158  ast_node *member = symbol_table_search_member(id, TYPE_NAME(prefix_type));
159 
160  if (!member) {
161  error_msg(node, "member '%s' does not exist in class '%s'", id,
162  TYPE_NAME(prefix_type));
163  return;
164  }
165 
166  // Copy the class definition of member in the identifier
167  IDENT_IN_CLASS(node) = IDENT_IN_CLASS(member);
168 
169  // Fill the member slot for further analyzes and type the node
170  IDENT_QUALIFIED(node) = member;
171  AST_TYPE(node) = AST_TYPE(member);
172 }

175 static void simple_identifier_analysis(ast_node *node, char *id){
176  // Try to see if "id" is a field of the current class.
177  // If so, add "this" as (an implicit prefix)
178  if (current_class) {
179  char * class_name = IDENT_VAL(CLASS_NAME(current_class));
180  ast_node *member = symbol_table_search_member(id, class_name);
181  ast_node *local = symbol_table_search_blocks(id);
182  if (!local && member) {
183  // add the prefix "this" to current object
184  IDENT_PREFIX(node) = make_this();
185  prefix_analysis(node, id); // A little bit overkill, but safer
186  return;
187  }
188  }
189  // Identifier is really a non qualified one
190  ast_node *var = symbol_table_search(id);
191 
192  AST_TYPE(node) = int_type; // for now
193  if (!var) {
194  // var is undeclared. Declare it as an integer to avoid further errors
195  symbol_table_declare_object(id, node);
196  error_msg(node, "%s is not declared", id);
197  }
198  else if (AST_KIND(var) == k_function) {
199  // var is declared but as a function.
200  error_msg(node, "%s function used as variable", id);
201  }
202  else if (AST_KIND(var) == k_class) {
203  // var is declared as a class
204  AST_TYPE(node) = AST_TYPE(var);
205  }
206  else if (AST_KIND(var) == k_identifier) {
207  // Set type of the node to the type of the identifier in the table
208  AST_TYPE(node) = AST_TYPE(var);
209  }
210  else {
211  die("fatal error when analyzing an identifier");
212  }
213 }

227 void analysis_type(ast_node *node){
228  if (TYPE_IS_STANDARD(node)) return; // Standard type are correct
229 
230  // Type is an ident. It must denote a class
231  ast_node *k = symbol_table_search(TYPE_NAME(node));
232 
233  if (!k) {
234  error_msg(node, "class '%s' is not declared", TYPE_NAME(node));
235  AST_TYPE(node) = NULL;
236  } else if (AST_KIND(k) != k_class) {
237  error_msg(node, "%s does not denote a class", TYPE_NAME(node));
238  AST_TYPE(node) = NULL;
239  } else
240  AST_TYPE(node) = k;
241 }

249 static int is_const(ast_node *node) {
251  if (AST_KIND(node) == k_constant)
252  return 1;
253  if (AST_KIND(node) == k_expression) {
254  int c1, c2=1, c3=1;
255  switch (EXPRESSION_ARITY(node)) {
256  case 3: c3 = is_const(EXPRESSION_OP3(node)); // no break !!!
257  case 2: c2 = is_const(EXPRESSION_OP2(node)); // no break !!!
258  case 1: c1 = is_const(EXPRESSION_OP1(node)); break;
259  default: die("weird operator arity");
260  }
261  return c1 && c2 && c3;
262  }
263  // If we are here, node doesn't denote a constant
264  return 0;
265 }

268 static void decl_locals(ast_node *node, ast_node *in_class) {
269  analysis(AST_TYPE(node));
270 
271  // Verify that we don't declare void vars/members
272  if (AST_TYPE(node) == void_type)
273  error_msg(node, "void is an improper type here");
274 
275  // Enter all the vars/members in the symbol table
276  bool global_decl = (!current_class && !current_function);
277 
278  FORLIST(p, VARDECL_VARS(node)) {
279  // p denote a couple with a variable identifier and initial value
280  ast_node *var = ((ast_node **)list_item_data(p))[0];
281  ast_node *val = ((ast_node **)list_item_data(p))[1];
282 
283  // If in_class is NULL we declare a variable. Otherwise, node denotes a
284  // member symbol which must be declared in the class hash table
285  if (in_class) {
286  symbol_table_declare_member(IDENT_VAL(var), var, in_class);
287  IDENT_IN_CLASS(var) = in_class;
288  }
289  else
290  symbol_table_declare_object(IDENT_VAL(var), var);
291 
292  // type the expression and verify that the type is conform to declaration
293  if (val) {
294  analysis(val);
295  if (!compatible_types(node, val))
296  error_msg(node, "initializer type incompatible with '%s'", IDENT_VAL(var));
297  if (global_decl && !is_const(val))
298  error_msg(node, "initializer for '%s' is not constant", IDENT_VAL(var));
299  }
300  }
301 }

317 void analysis_class(ast_node *node) {
318  ast_node *klass = CLASS_NAME(node);
319  ast_node *parent = NULL;
320  List members = CLASS_MEMBERS(node);
321  char *name = IDENT_VAL(klass);
322 
323  // Search the given name in the symbol table
324  ast_node *prev = symbol_table_search(name);
325 
326  // Verify that class was not already defined
327  if (prev) {
328  if (AST_KIND(prev) == k_class) {
329  if (CLASS_MEMBERS(prev))
330  error_msg(node, "class '%s' was already defined", name);
331  } else
332  error_msg(node, "class '%s' cannot be defined (symbol already used)", name);
333  return;
334  }
335 
336  // Class was not defined before
337  if (CLASS_PARENT(node)) {
338  // Verify that the parent class exists
339  char *pname = IDENT_VAL(CLASS_PARENT(node));
340 
341  parent = symbol_table_search(pname);
342 
343  if (!parent)
344  error_msg(node, "parent class '%s' does not exist", pname);
345  else if (AST_KIND(parent) != k_class) {
346  error_msg(node, "%s is not a class", pname);
347  parent = NULL; // to continue analysis further
348  } else {
349  // Correct the parent with the class (instead of its name)
350  free_node(CLASS_PARENT(node)); // it was a simple identifier
351  CLASS_PARENT(node) = parent; // Set the parent to the class itself instead
352  }
353  }
354 
355  if (!parent)
356  // No class was givent (or error detected) => parent is "Object"
357  CLASS_PARENT(node) = parent = Object_class;
358 
359  // declare the class and its members
360  symbol_table_declare_class(name, node, parent);
361  if (members) {
362  current_class = node;
363  FORLIST(p, CLASS_MEMBERS(node)) {
364  ast_node *member = list_item_data(p);
365 
366  if (AST_KIND(member) == k_var_decl)
367  decl_locals(member, node); // member is a field list
368  else
369  analysis_function(member); // member is a method
370  }
371 
372  // Build the class vtable
373  build_class_vtable(current_class);
374 
375  current_class = NULL;
376  }
377 }

379 // ---- Creation of the class vtable ------------------------------
380 static void build_class_vtable_unsorted(ast_node *klass, List lst) {
381  if (klass) {
382  // Recurse
383  build_class_vtable_unsorted(CLASS_PARENT(klass), lst);
384  FORLIST(p, CLASS_MEMBERS(klass)) {
385  if (AST_KIND(list_item_data(p)) == k_function) {
386  list_append(lst, list_item_data(p), NULL);
387  }
388  }
389  }

392 static void build_class_vtable(ast_node *klass) {
393  List tmp = list_create();
394  int index = 0;
395 
396  build_class_vtable_unsorted(klass, tmp);
397  // We had running down the hierarchy and tmp may contain duplicates.
398  // Build a sorted set of methods
399  List vtable = list_create();
400 
401  FORLIST(p, tmp) {
402  ast_node *method = list_item_data(p);
403  char *method_name = IDENT_VAL(FUNCTION_NAME(method));
404  ast_node *last = NULL;
405  int occurence = 0;
406 
407  FORLIST(q, tmp) {
408  ast_node *candidate = list_item_data(q);
409 
410  if (strcmp(method_name, IDENT_VAL(FUNCTION_NAME(candidate))) == 0) {
411  last = candidate;
412  occurence += 1;
413  }
414  }
415  if ((occurence == 1) || (occurence > 1 && method != last)) {
416  FUNCTION_INDEX(last) = index++;
417  list_append(vtable, last, NULL);
418  }
419  }
420 
421  list_destroy(tmp);
422  CLASS_VTABLE(klass) = vtable;
423 }

456 void analysis_expression(ast_node *node) {
457  char *name = EXPRESSION_OPERATOR(node);
458  ast_node *op1 = EXPRESSION_OP1(node);
459  ast_node *op2 = EXPRESSION_OP2(node); /* may be NULL */
460  ast_node *op3 = EXPRESSION_OP3(node); /* may be NULL */
461 
462  switch (EXPRESSION_ARITY(node)) {
463  case 1: analysis(op1);
464  switch(EXPRESSION_KIND(node)) {
465  case preincr:
466  case postincr:
467  case uarith:
468  if (!is_number(op1))
469  error_msg(node, "operand of `%s' must be a number", name);
470  AST_TYPE(node) = int_type;
471  break;
472  case ulogic:
473  if (AST_TYPE(op1) != bool_type)
474  error_msg(node, "operand of 'not' must be bool");
475  AST_TYPE(node) = bool_type;
476  break;
477  case parenthesis:
478  AST_TYPE(node) = AST_TYPE(op1);
479  break;
480  default: die("unhandled unary expression");
481  }
482  break;
483 
484  case 2: analysis(op1); analysis(op2);
485  if (!AST_TYPE(op1) || !AST_TYPE(op2)) //
486  // if one of the types is NULL, we consider that this is compatible
487  // It arises generally when a variable is auto-declared
488  break;
489 
490  // Special test for isa
491  if (EXPRESSION_KIND(node) == isa) {
492  if (TYPE_IS_STANDARD(AST_TYPE(op1))) {
493  error_msg(node, "first operand of instanceof is not an object");
494  }
495  if (TYPE_IS_STANDARD(AST_TYPE(op2))) {
496  error_msg(node, "second operand of instanceof is not an object");
497  }
498 
499  AST_TYPE(node) = bool_type;
500  break;
501  }
502 
503  if (! compatible_types(op1, op2))
504  error_msg(node, "operands of %s have incompatible types",
505  !strcmp(name, "=")? "assignment" : name);
506 
507  switch(EXPRESSION_KIND(node)) {
508  case assign:
509  AST_TYPE(node)= AST_TYPE(op2);
510  break;
511  case barith:
512  if (!is_number(op1) || !is_number(op2))
513  error_msg(node, "operands of %s must be numbers", name);
514  AST_TYPE(node) = (AST_TYPE(op1)== float_type ||
515  AST_TYPE(op2) == float_type)? float_type: int_type;
516  break;
517  case blogic:
518  if (AST_TYPE(op1) != bool_type || AST_TYPE(op2) != bool_type)
519  error_msg(node, "operands of %s must be bool", name);
520  AST_TYPE(node) = bool_type;
521  break;
522  case comp:
523  AST_TYPE(node) = bool_type;
524  break;
525  default: die("unhandled binary expression");
526  }
527  break;
528  case 3: // the only ternary operator is expr ? expr : expr
529  analysis(op1);analysis(op2); analysis(op3);
530  if (AST_TYPE(op1) != bool_type)
531  error_msg(node, "test is not boolean in a '?:' expression");
532  else {
533  if (AST_TYPE(op2) != AST_TYPE(op3))
534  // NOTE: en fait c'est bien plus difficle que Ã§a....
535  error_msg(node, "branches of '?:' are of different types");
536  }
537  AST_TYPE(node) = AST_TYPE(op2);
538  break;
539  }
540 }

548 static int parameter_cmp(List formal, List effective, char *id, ast_node *node) {
549  List_item p1 = list_head(formal);
550  List_item p2 = list_head(effective);
551  int err = 0;
552 
553  for (int idx=1; p1 && p2; p1 = list_item_next(p1), p2 = list_item_next(p2), idx++){
554  if (!compatible_types(list_item_data(p1), list_item_data(p2))) {
555  error_msg(node, "type of parameter #%d is not compatible with "
556  "previous definition of '%s'", idx, id);
557  err += 1;
558  }
559  }
560 
561  if (p1 && !p2) {
562  error_msg(node, "call or definition of '%s' has not enough parameters", id);
563  err++;
564  }
565  if (!p1 && p2) {
566  error_msg(node, "call or definition of '%s' has too much parameters", id);
567  err++;
568  }
569  return !err;
570 }

573 static void analysis_formal(List formal) {
574  int pos = 0, len = list_size(formal);
575  char *names[len];
576 
577  FORLIST(p, formal) {
578  ast_node *param= list_item_data(p);
579 
580  // verify formal parameter type
581  if (AST_TYPE(param) == void_type)
582  error_msg(param, "type of parameter cannot be void");
583  else
584  analysis(AST_TYPE(param));
585 
586  // Verify that this parameter was not previously used
587  for (int i = 0; i < pos; i ++) {
588  if (strcmp(IDENT_VAL(param), names[i]) == 0) {
589  error_msg(param, "name '%s' was already used for paramater #%d",
590  IDENT_VAL(param), i+1);
591  break;
592  }
593  }
594  // Retain the current parameter name for further controls
595  names[pos++] = IDENT_VAL(param);
596  }
597 }

602 static void analysis_function_body(ast_node *node) {
603  struct s_function *n = (struct s_function *) node;
604 
605  // Declare the parameters in a new scope
606  enter_scope();
607  FORLIST(p, n->parameters) {
608  ast_node *param= list_item_data(p);
609  symbol_table_declare_object(IDENT_VAL(param), param);
610  }
611 
612  // Analyze the statements of the function (don't use analysis_block_statement
613  // which enter a new scope)
614  struct s_block_statement *body = (struct s_block_statement *) n->body;
615 
616  current_function = node;
617  FORLIST(p, body->statements) analysis(list_item_data(p));
618  current_function = NULL;
619 
620  leave_scope();
621 }

624 void analysis_function(ast_node *node) { // Used also for methods
625  struct s_function *n = (struct s_function *) node;
626  char *id = IDENT_VAL(n->name);
627 
628  n->in_class = current_class; // Set the embedding class (if any)
629  analysis(AST_TYPE(node)); // Verify the return type
630  analysis_formal(FUNCTION_PARAMETERS(node)); // Verify the formal parameter list
631 
632  // Declare the function/method
633  if (current_class) { // **** This is a METHOD declaration
634  char *kid = IDENT_VAL(CLASS_NAME(current_class));
635  bool error = false;
636  ast_node *prev = symbol_table_search_member(id, kid);
637 
638  if (prev) {
639  if (FUNCTION_IN_CLASS(prev) == current_class) {
640  // Verify that the previous method was not declared in the current class
641  error_msg(node, "method %s was already declared in this class", id);
642  error = true;
643  } else if (AST_KIND(prev) != k_function) {
644  // Verify that previous declaration was not a field
645  error_msg(node, "inherited %s was not declared as a method", id);
646  error = true;
647  } else if (! parameter_cmp(FUNCTION_PARAMETERS(node),FUNCTION_PARAMETERS(prev),
648  id,node)){
649  // Verify that previous and current prototypes are compatible
650  error_msg(node, "parameters of %s incompatible with previous declaration", id);
651  error = true;
652  }
653  else if (! compatible_types(prev, node)) {
654  // Verify that return type of current method is compatible with inherited one
655  error_msg(node, "method return type is not compatible with overridden method");
656  error = true;
657  }
658  }
659  if (!error) // No error â declare the method in the current class
660  symbol_table_declare_member(id, node, current_class);
661  }
662  else { // **** This is a FUNCTION declaration
663  ast_node *prev = symbol_table_search(id);
664  if (prev && (AST_KIND(prev) == k_function) && (FUNCTION_BODY(prev) == NULL)) {
665  // prev was a prototype
666  parameter_cmp(FUNCTION_PARAMETERS(prev),FUNCTION_PARAMETERS(node),id,node);
667  if (strcmp(TYPE_NAME(AST_TYPE(prev)), TYPE_NAME(AST_TYPE(node))) != 0)
668  error_msg(node, "function return type is different from '%s' prototype", id);
669  // Keep the new definition rather than old prototype
670  free_node(prev);
671  symbol_table_delete(id);
672  }
673  symbol_table_declare_object(id, node);
674  }
675 
676  // Body analysis
677  if (n->body) analysis_function_body(node);
678 }

684 static void analysis_method_call(ast_node *node) {
685  struct s_call *n = (struct s_call *) node;
686  ast_node *callee = n->callee;
687 
688  // Type the callee and the parameter list
689  analysis(callee);
690  FORLIST(p, n->parameters) analysis(list_item_data(p));

692  if (!AST_TYPE(callee)) {
693  // Error in previous analysis
694  AST_TYPE(node) = NULL;
695  return;
696  }

704  TYPE_NAME(AST_TYPE(IDENT_PREFIX(callee))));
705  else {
706  parameter_cmp(FUNCTION_PARAMETERS(func), CALL_PARAMETERS(n), id, node);
707  AST_TYPE(node) = AST_TYPE(callee);
708  }

712 static void analysis_function_call(ast_node *node) {
713  struct s_call *n = (struct s_call *) node;
714 
715  // Simple function call: Search function name
716  char *id = IDENT_VAL(n->callee);
717  struct ast_node *func = symbol_table_search(id);
718 
719  if (!func || (func && AST_KIND(func) != k_function))
720  error_msg(node, "function %s is undefined", id);
721  // if !func, we could declare it. Refrain to do so: which type?, which params?
722  else
723  // func is declared as a function. Set type of node to its return type
724  AST_TYPE(n) = AST_TYPE(func);
725 
726  // Type the parameter list
727  FORLIST(p, n->parameters) analysis(list_item_data(p));
728 
729  if (func)
730  // Compare formal parameter list with effective list
731  parameter_cmp(FUNCTION_PARAMETERS(func), CALL_PARAMETERS(n), id, node);
732 }
