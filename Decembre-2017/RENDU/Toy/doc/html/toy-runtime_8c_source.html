158 void _toy_read_bool(char *o) {
159  *o = 0; // Set result to false by default
160 
161  if (!feof(stdin)) {
162  char buffer[READ_BUFF_SIZE];
163 
164  if (fscanf(stdin, " %[01a-zA-Z]s", buffer) == 1) {
165  if ((strcasecmp(buffer, "true")==0) ||
166  (strcasecmp(buffer, "t")==0) ||
167  (strcmp(buffer, "1")==0))
168  *o = 1;
169  if (!feof(stdin)) fgetc(stdin); // to skip the terminator
170  }
171  }
172 }

174 void _toy_read_string(_toy_string *o) {
175  *o = NULL; // set result to "null" by default
176 
177  if (!feof(stdin)) {
178  char buffer[READ_BUFF_SIZE];
179 
180  if (fgets(buffer, READ_BUFF_SIZE, stdin)) {
181  int len = strlen(buffer);
182  // Suppress a possible final newline
183  if (buffer[len-1] == '\n') buffer[len-1] = '\0';
184  *o = strdup(buffer);
185  }
186  }
187 }
