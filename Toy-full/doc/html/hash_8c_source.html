53 static unsigned long hash(const char *str)
54 {
55  // djb2: this function was first reported by Dan Bernstein many
56  // years ago in comp.lang.c
57  unsigned long hash = 5381;
58  for ( ; *str; str++)
59  hash = ((hash << 5) + hash) + *str; // hash*33 + *str
60  return hash;
61 }

78 static void enlarge_table(Hash_table h)
79 {
80  int new_size = h->size * GROWTH_FACTOR;
81  struct elt **new_table = calloc(sizeof(struct elt), new_size);
82 
83  // printf("Enlarge table %p from %d to %d\n", h, h->size, new_size);
84 
85  /* store all the elements of the old table in the new one */
86  for(int i = 0; i < h->size; i++) {
87  struct elt *e = h->table[i];
88 
89  while (e) {
90  unsigned long bucket_index = hash(e->key) % new_size;
91  struct elt *next = e->next;
92 
93  e->next = new_table[bucket_index];
94  new_table[bucket_index] = e;
95  e = next;
96  }
97  }
98 
99  /* Free the old table */
100  free(h->table);
101 
102  /* keep the new table */
103  h->size = new_size;
104  h->table = new_table;
105 }

138 void hash_table_add(Hash_table h, const char *key, void *value, void (*free)(void *))
139 {
140  struct elt *e;
141  unsigned long bucket_index = hash(key) % h->size;
142 
143  /* Create a new element */
144  e = malloc(sizeof(struct elt));
145  e->key = strdup(key);
146  e->value = value;
147  e->free = free;
148 
149  /* enter it in the table */
150  e->next = h->table[bucket_index];
151  h->table[bucket_index] = e;
152 
153 
154  h->n += 1;
155 
156  /* If the table seem to have not enough room, enlarge it */
157  if(h->n >= (h->size * MAX_LOAD_FACTOR))
158  enlarge_table(h);

174 void hash_table_delete(Hash_table h, const char *key)
175 {
176  struct elt *e, *prev;
177  unsigned long bucket_index = hash(key) % h->size;
178 
179  for(prev=NULL, e=h->table[bucket_index]; e; prev=e, e=e->next) {
180  if (strcmp(e->key, key) == 0) {
181  if (prev)
182  prev->next = e->next;
183  else
184  h->table[bucket_index] = e->next;
185 
186  free(e->key);
187  if (e->free) e->free(e->value);
188  free(e);
189  return;
190  }
191  }
