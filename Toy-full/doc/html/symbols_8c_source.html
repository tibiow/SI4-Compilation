97 void symbol_table_declare_object(char *id, ast_node *obj) {

99 
100  if (old)

108 //
109 // Declare a new class.
110 //
111 void symbol_table_declare_class(char *id, ast_node *obj, ast_node *parent) {
112  Symbol_table st;
113 
114  // Declare the new symbol (we know that it is absent, could be simplified)
115  symbol_table_declare_object(id, obj);
116 
117  // Allocate a new symbol table for the members of the class
118  st = must_malloc(sizeof(struct symbol_table));
119 
120  st->table = hash_table_create();
121  st->next = parent ? hash_table_search(all_classes, IDENT_VAL(CLASS_NAME(parent))):
122  NULL;
123  hash_table_add(all_classes, id, st, NULL);
124 }

128 // Declare a class member

130 void symbol_table_declare_member(char *id, ast_node *obj, ast_node *klass) {
131  char *class_id = IDENT_VAL(CLASS_NAME(klass));;
132 
133  // Find the table to search in
134  Symbol_table ht = (Symbol_table) hash_table_search(all_classes, class_id);
135 
136  if (!ht) die("we have a serious problem: no hash table for the current class");
137 
138  // Declare the symbol
139  hash_table_add(ht->table , id, obj, NULL); // class members will never be freed
140 }

143 //
144 // Search a symbol starting from current scope.
145 //
146 ast_node *symbol_table_search(char *id) {
147  Symbol_table t;
148 
149  for (t = current_table; t; t = t->next) {
150  ast_node *res= hash_table_search(t->table, id);
151  if (res) return res;
152  }
153  return NULL;

170 //
171 // Search a member of given class.
172 //
173 ast_node *symbol_table_search_member(char *id, char *class_id) {
174  // Find the table to search in
175  Symbol_table ht = (Symbol_table) hash_table_search(all_classes, class_id);
176 
177  if (!ht)
178  die("we have a serious problem: no hash table for the current class");
179 
180  // Search recursively the symbol in the class hash table
181  for (Symbol_table t = ht; t; t = t->next) {
182  ast_node *old = hash_table_search(t->table, id);
183 
184  if (old)
185  return old;
186  }
187  return NULL;
188 }

196 //
197 // Free the old (no more used) tables.
198 //
199 void symbol_table_free_unused_tables(void) {
200  Symbol_table p, next;
201 
202  for (p = tables_to_free; p ; p = next) {
203  next = p->next;
204  hash_table_destroy(p->table);
205  free(p);
206  }
207  tables_to_free = NULL;
208 }
