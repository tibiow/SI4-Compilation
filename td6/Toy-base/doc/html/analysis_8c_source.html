49 static bool compatible_types(ast_node *e1, ast_node *e2) {
50  ast_node *t1 = AST_TYPE(e1), *t2 = AST_TYPE(e2);
51 
52  if (!t1 || !t2) return true; // to avoid error cascades => true
53  if (t1 == t2) return true; // same types => true
54 
55  return false;
56 }

71 void analysis_identifier(ast_node *node){
72  char *id = IDENT_VAL(node);
73  ast_node *var = symbol_table_search(id);
74 
75  AST_TYPE(node) = int_type; // for now
76  if (!var) {
77  // var is undeclared. Declare it as an integer to avoid further errors
78  symbol_table_declare_object(id, node);
79  error_msg(node, "%s is not declared", id);
80  }
81  else if (AST_KIND(var) == k_function) {
82  // var is declared but as a function.
83  error_msg(node, "%s function used as variable", id);
84  }
85  else if (AST_KIND(var) == k_identifier) {
86  // Set type of the node to the type of the identifier in the table
87  AST_TYPE(node) = AST_TYPE(var);
88  }
89  else {
90  die("fatal error when analyzing an identifier");
91  }
92 }

110 static int is_const(ast_node *node) {
112  if (AST_KIND(node) == k_constant)
113  return 1;
114  if (AST_KIND(node) == k_expression) {
115  int c1, c2=1, c3=1;
116  switch (EXPRESSION_ARITY(node)) {
117  case 3: c3 = is_const(EXPRESSION_OP3(node)); // no break !!!
118  case 2: c2 = is_const(EXPRESSION_OP2(node)); // no break !!!
119  case 1: c1 = is_const(EXPRESSION_OP1(node)); break;
120  default: die("weird operator arity");
121  }
122  return c1 && c2 && c3;
123  }
124  // If we are here, node doesn't denote a constant
125  return 0;
126 }

129 void analysis_var_decl(ast_node *node) {
130  analysis(AST_TYPE(node));
131 
132  // Verify that we don't declare void vars
133  if (AST_TYPE(node) == void_type)
134  error_msg(node, "void is an improper type here");
135 
136  // Enter all the vars in the symbol table
137  bool global_decl = (!current_function);
138 
139  FORLIST(p, VARDECL_VARS(node)) {
140  // p denote a couple with a variable identifier and initial value
141  ast_node *var = ((ast_node **)list_item_data(p))[0];
142  ast_node *val = ((ast_node **)list_item_data(p))[1];
143 
144  symbol_table_declare_object(IDENT_VAL(var), var);
145 
146  // type the expression and verify that the type is conform to declaration
147  if (val) {
148  analysis(val);
149  if (!compatible_types(node, val))
150  error_msg(node, "initializer type incompatible with '%s'", IDENT_VAL(var));
151  if (global_decl && !is_const(val))
152  error_msg(node, "initializer for '%s' is not constant", IDENT_VAL(var));
153  }
154  }
155 }

163 void analysis_expression(ast_node *node) {
164  char *name = EXPRESSION_OPERATOR(node);
165  ast_node *op1 = EXPRESSION_OP1(node);
166  ast_node *op2 = EXPRESSION_OP2(node); /* may be NULL */
167  ast_node *op3 = EXPRESSION_OP3(node); /* may be NULL */
168 
169  switch (EXPRESSION_ARITY(node)) {
170  case 1: analysis(op1);
171  switch(EXPRESSION_KIND(node)) {
172  case uarith:
173  if (AST_TYPE(op1) != int_type)
174  error_msg(node, "operand of unary %s must be int", name);
175  AST_TYPE(node) = int_type;
176  break;
177  case ulogic:
178  if (AST_TYPE(op1) != bool_type)
179  error_msg(node, "operand of unary %s must be bool", name);
180  AST_TYPE(node) = bool_type;
181  break;
182  case parenthesis:
183  AST_TYPE(node) = AST_TYPE(op1);
184  break;
185  default: die("unhandled unary expression");
186  }
187  break;
188 
189  case 2: analysis(op1); analysis(op2);
190  if (!AST_TYPE(op1) || !AST_TYPE(op2)) //
191  // if one of the types is NULL, we consider that this is compatible
192  // It arises generally when a variable is auto-declared
193  break;
194  if (! compatible_types(op1, op2))
195  error_msg(node, "operands of %s have incompatible types",
196  !strcmp(name, "=")? "assignment" : name);
197 
198  switch(EXPRESSION_KIND(node)) {
199  case assign:
200  AST_TYPE(node)= AST_TYPE(op2);
201  break;
202  case barith:
203  if (AST_TYPE(op1) != int_type || AST_TYPE(op2) != int_type)
204  error_msg(node, "operands of %s must be int", name);
205  AST_TYPE(node) = int_type;
206  break;
207  case blogic:
208  if (AST_TYPE(op1) != bool_type || AST_TYPE(op2) != bool_type)
209  error_msg(node, "operands of %s must be bool", name);
210  AST_TYPE(node) = bool_type;
211  break;
212  case comp:
213  AST_TYPE(node) = bool_type;
214  break;
215  default: die("unhandled binary expression");
216  }
217  break;
218  case 3: // the only ternary operator is expr ? expr : expr
219  analysis(op1);analysis(op2); analysis(op3);
220  if (AST_TYPE(op1) != bool_type)
221  error_msg(node, "test is not boolean in a '?:' expression");
222  else {
223  if (AST_TYPE(op2) != AST_TYPE(op3))
224  // FIXME: c'est bien plus difficle que ça....
225  error_msg(node, "branches of '?:' are of different types");
226  }
227  AST_TYPE(node) = AST_TYPE(op2);
228  break;
229  }
230 }

238 static int parameter_cmp(List formal, List effective, char *id, ast_node *node) {
239  List_item p1 = list_head(formal);
240  List_item p2 = list_head(effective);
241  int err = 0;
242 
243  for (int idx=1; p1 && p2; p1 = list_item_next(p1), p2 = list_item_next(p2), idx++){
244  if (!compatible_types(list_item_data(p1), list_item_data(p2))) {
245  error_msg(node, "type of parameter #%d is not compatible with "
246  "previous definition", idx);
247  err += 1;
248  }
249  }
250 
251  if (p1 && !p2) {
252  error_msg(node, "call or definition of %s has not enough parameters", id);
253  err++;
254  }
255  if (!p1 && p2) {
256  error_msg(node, "call or definition of %s has too much parameters", id);
257  err++;
258  }
259  return !err;
260 }

263 static void analysis_formal(List formal) {
264  int pos = 0, len = list_size(formal);
265  char *names[len];
266 
267  FORLIST(p, formal) {
268  ast_node *param= list_item_data(p);
269 
270  // verify formal parameter type
271  if (AST_TYPE(param) == void_type)
272  error_msg(param, "type of parameter cannot be void");
273  else
274  analysis(AST_TYPE(param));
275 
276  // Verify that this parameter was not previously used
277  for (int i = 0; i < pos; i ++) {
278  if (strcmp(IDENT_VAL(param), names[i]) == 0) {
279  error_msg(param, "name '%s' was already used for paramater #%d",
280  IDENT_VAL(param), i+1);
281  break;
282  }
283  }
284  // Retain the current parameter name for further controls
285  names[pos++] = IDENT_VAL(param);
286  }
287 }

292 static void analysis_function_body(ast_node *node) {
293  struct s_function *n = (struct s_function *) node;
294 
295  // Declare the parameters in a new scope
296  enter_scope();
297  FORLIST(p, n->parameters) {
298  ast_node *param= list_item_data(p);
299 
300  symbol_table_declare_object(IDENT_VAL(param), param);
301  }
302 
303  // Analyze the statements of the function (don't use analysis_block_statement
304  // which enter a new scope)
305  struct s_block_statement *body = (struct s_block_statement *) n->body;
306 
307  current_function = node;
308  FORLIST(p, body->statements) analysis(list_item_data(p));
309  current_function = NULL;
310 
311  leave_scope();
312 }

315 void analysis_function(ast_node *node) {
316  struct s_function *n = (struct s_function *) node;
317  char *id = IDENT_VAL(n->name);
318 
319  analysis(AST_TYPE(node)); // Verify the return type
320  analysis_formal(FUNCTION_PARAMETERS(node)); // Verify the formal parameter list
321 
322  // Declare the function
323  ast_node *proto = symbol_table_search_prototype(id);
324  if (proto) {
325  if (parameter_cmp(FUNCTION_PARAMETERS(proto),FUNCTION_PARAMETERS(node),id,node))
326  symbol_table_replace_prototype(id, node);
327  } else
328  symbol_table_declare_object(id, node);
329 
330  // Body analysis
331  if (n->body) analysis_function_body(node);
332 }

338 void analysis_call(ast_node *node) {
339  struct s_call *n = (struct s_call *) node;
340 
341  // Simple function call: Search function name
342  char *id = IDENT_VAL(n->callee);
343  struct ast_node *func = symbol_table_search(id);
344 
345  if (!func || (func && AST_KIND(func) != k_function))
346  error_msg(node, "function %s is undefined", id);
347  // if !func, we could declare it. Refrain to do so: which type?, which params?
348  else
349  // func is declared as a function. Set type of node to its return type
350  AST_TYPE(n) = AST_TYPE(func);
351 
352  // Type the parameter list
353  FORLIST(p, n->parameters) analysis(list_item_data(p));
354 
355  if (func)
356  // Compare formal parameter list with effective list
357  parameter_cmp(FUNCTION_PARAMETERS(func), CALL_PARAMETERS(n), id, node);
358 }
