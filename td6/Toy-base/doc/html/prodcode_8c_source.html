122 static void produce_declarations(ast_node *node, int emit_init) {
124  struct s_var_decl *n = (struct s_var_decl *) node;
125  int len = list_size(n->vars);
126 
127  code(AST_TYPE(node)); emit(" ");
128  FORLIST(p, n->vars) {
129  ast_node *var = ((ast_node **)list_item_data(p))[0];
130  ast_node *val = ((ast_node **)list_item_data(p))[1];
131 
132  code(var);
133  if (emit_init) { // Initialization code must be issued
134  emit(" = ");
135  if (val) code_expr_cast(val); else emit(TYPE_DEFAULT_INIT(AST_TYPE(node)));
136  }
137  if (--len) emit(", ");
138  }
139  emit(";\n");
140 }

152 void produce_code_expression(ast_node *node) {
153  char *name = EXPRESSION_OPERATOR(node);
154 
155  switch (EXPRESSION_ARITY(node)) {
156  case 1:
157  switch(name[0]) {
158  case '-':
159  case '!':
160  emit(name); code(EXPRESSION_OP1(node));
161  break;
162  case '(':
163  emit("("); code(EXPRESSION_OP1(node)); emit(")");
164  break;
165  }
166  break;
167  case 2:
168  code(EXPRESSION_OP1(node));
169  emit(" %s ", name);
170  code_expr_cast(EXPRESSION_OP2(node));
171  break;
172  case 3: /* the only ternary operator is expr ? expr : expr */
173  code(EXPRESSION_OP1(node)); emit (" ? "); code(EXPRESSION_OP2(node));
174  emit(" : "); code(EXPRESSION_OP3(node));
175  break;
176  }
177 }

184 void produce_code_function(ast_node *node) {
185  struct s_function *n = (struct s_function *) node;
186  int comma = 0;
187 
188  // Emit function type and name
189  code(AST_TYPE(node)); emit(" "); emit_function_name(node);
190 
191  // Produce parameters
192  emit("(");
193  if (list_size(n->parameters) == 0)
194  emit("void");
195  else {
196  FORLIST(p, n->parameters) {
197  ast_node *v = list_item_data(p);
198  if (comma++) emit(", ");
199  code(AST_TYPE(v)); emit(" "); code(v);
200  }
201  }
202  emit(")");
203 
204  // Body or ';'
205  if (n->body) { emit(" "); code(n->body); } else emit (";\n");
206 }

209 void produce_code_call(ast_node *node) {
210  struct s_call *n = (struct s_call *) node;
211  int comma = 0;
212 
213  if (!AST_TYPE(node)) return; // previous error detected
214 
215  code(n->callee); emit("(");
216  // produce the parameters
217  FORLIST(p, n->parameters) {
218  if (comma++) emit(", ");
219  code_expr_cast(list_item_data(p));
220  }
221  emit(")");
222 }

270 void produce_code_print_statement(ast_node *node) {
271  struct s_print_statement *n = (struct s_print_statement *) node;
272  bool first_parameter = true;
273 
274  // Note: all the print must be in the same block (think to 'if (C) print(a, b)'
275  emit("{\n"); indent(+1);
276  FORLIST(p, n->parameters) {
277  ast_node *item = list_item_data(p);
278 
279  if (! first_parameter) indent(0);
280  if (AST_TYPE(item) == int_type) { // int
281  emit("printf(\"%%d\", ");
282  } else if (AST_TYPE(item) == bool_type) { // bool
283  emit("_toy_print_bool(");
284  } else if (AST_TYPE(item) == string_type) { //string
285  emit("printf(\"%%s\", ");
286  } else { // object
287  emit("_toy_print_object((Object) ");
288  }
289  code(item);
290  emit(");\n");
291  first_parameter = false;
292  }
293  indent(0); emit("fflush(stdout);\n");
294  indent(-1); emit("}\n");
295 }
