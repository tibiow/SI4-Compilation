98 //
99 // Search a symbol starting from current scope.
100 //
101 ast_node *symbol_table_search(char *id) {
102  Symbol_table t;
103 
104  for (t = current_table; t; t = t->next) {
105  ast_node *res= hash_table_search(t->table, id);
106  if (res) return res;
107  }
108  return NULL;

112 //
113 // Search a prototype in the current scope.
114 //
115 ast_node *symbol_table_search_prototype(char *id) {
116  ast_node *old = hash_table_search(current_table->table, id);
117 
118  if (old && (AST_KIND(old)==k_function) && (FUNCTION_BODY(old)==NULL))
119  return old; // old is a prototype
120  return NULL;
121 }

127 void symbol_table_replace_prototype(char *id, ast_node *func) {
128  hash_table_delete(current_table->table, id);
