171 static void free_one_var_decl(void *p) {
172  // p is a two cell array where p[0] is the var and p[1] its init. expression
173  ast_node **table = p;
174 
175  // Free the expression (not the var which must stay in symbol table
176  free_node(table[1]);
177 }

179 ast_node *make_var_decl(ast_node *var, ast_node *type, ast_node* expr) {
180  DEF_AST(p, var_decl, type);
181 
182  /* add the given variable to a newly created list */
183  p->vars = list_create();
184  add_variable_to_decl((ast_node *) p, var, expr);
185 
186  return (ast_node *) p;
187 }

189 void add_variable_to_decl(ast_node *decl, ast_node *var, ast_node *expr) {
190  ast_node **table = must_malloc(2 * sizeof(ast_node *));
191 
192  AST_TYPE(var) = AST_TYPE(decl);
193  // Build a couple for the var and its initial value (maybe NULL)
194  table[0] = var;
195  table[1] = expr;
196  list_append(VARDECL_VARS(decl), table, free_one_var_decl);
197 }

254 ast_node *make_expression(char *operator, enum expr_kind ekind, int arity, ...)
255 {
256  va_list ap;
257  va_start(ap, arity);
258 
259  DEF_AST(p, expression, NULL);
260  p->operator = operator;
261  p->cast = NULL; // NULL => no cast needed
262  p->ekind = ekind;
263  p->arity = arity;
264 
265  // Fill in the operands
266  EXPRESSION_OP1(p) = ((arity >= 1) ? va_arg(ap, ast_node *): NULL);
267  EXPRESSION_OP2(p) = ((arity >= 2) ? va_arg(ap, ast_node *): NULL);
268  EXPRESSION_OP3(p) = ((arity >= 3) ? va_arg(ap, ast_node *): NULL);
269  va_end(ap);
270 
271  return (ast_node *) p;
272 }
