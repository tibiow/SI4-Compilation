140 static void produce_declarations(ast_node *node, bool emit_init) {
143  struct s_var_decl *n = (struct s_var_decl *) node;
144  int len = list_size(n->vars);
145 
146  code(AST_TYPE(node)); emit(" ");
147  FORLIST(p, n->vars) {
148  ast_node *var = ((ast_node **)list_item_data(p))[0];
149  ast_node *val = ((ast_node **)list_item_data(p))[1];
150 
151  code(var);
152  if (emit_init) { // Initialization code must be issued
153  emit(" = ");
154  if (val) code_expr_cast(val); else emit(TYPE_DEFAULT_INIT(AST_TYPE(node)));
155  }
156  if (--len) emit(", ");
157  }
158  emit(";\n");
159 }

186  emit("typedef struct %s *%s; // class %s implementation typedef\n", name,name,name);
187  emit("struct %s {\n", name);

196 static void produce_class_constructor(ast_node *node, char *name, List members) {
197  // Emit function header
198  emit("%s _init_%s(%s this) { // %s class constructor\n", name, name, name, name);
199 
200  // Emit a call to the super class initialization
201  char *pname = IDENT_VAL(CLASS_NAME(CLASS_PARENT(node)));
202 
203  // Emit initialization of the super class and of the instance_of field
204  emit(" _init_%s((%s) this);\n", pname, pname);
205  emit(" this->_instance_of = &_toy_meta_%s;\n\n", name);
206 
207  // Emit direct members initialization
208  FORLIST(p, members) {
209  if (AST_KIND(list_item_data(p)) == k_var_decl) {
210  struct s_var_decl *n = list_item_data(p);
211 
212  FORLIST(p, n->vars) {
213  ast_node *var = ((ast_node **)list_item_data(p))[0];
214  ast_node *val = ((ast_node **)list_item_data(p))[1];
215 
216  indent(+1);
217  emit("this->"); code(var); emit(" = ");
218  if (val) code(val); else emit(TYPE_DEFAULT_INIT(AST_TYPE(n)));
219  emit(";\n");
220  indent(-1);
221  }
222  }
223  }
224 
225  // Emit the end of the constructor
226  emit(" return this;\n}\n\n");
227 }

258 void produce_code_class(ast_node *node) {
259  char *name = IDENT_VAL(CLASS_NAME(node));
260  List members = CLASS_MEMBERS(node);
261 
262  if (!members) {
263  // We just have a class declaration (without definition)
264  emit("typedef struct %s *%s;\n", name, name);
265  return;
266  }
267  // We have a complete definition of the class
268  emit("// ----------------------------------------\n");
269  emit("// Class %s implementation\n", name);
270  emit("// ----------------------------------------\n\n");
271  produce_class_struct(node, name);
272  produce_class_constructor(node, name, members);
273  produce_class_methods(node, name, members);
274  produce_class_metaclass(node, name);
275  emit("// ---- End of class %s\n\n", name);
276 }

295 void produce_code_expression(ast_node *node) {
296  char *name = EXPRESSION_OPERATOR(node);
297 
298  switch (EXPRESSION_ARITY(node)) {
299  case 1:
300  if (EXPRESSION_KIND(node) == preincr) {
301  emit(name); code(EXPRESSION_OP1(node));
302  } else if (EXPRESSION_KIND(node) == postincr) {
303  code(EXPRESSION_OP1(node)); emit(name);
304  } else {
305  switch(name[0]) {
306  case '-':
307  case '!':
308  emit("%s", name);
309  // NO BREAK HERE !!!!
310  case '(':
311  emit("("); code(EXPRESSION_OP1(node)); emit(")");
312  break;
313  }
314  }
315  break;
316  case 2:
317  if (EXPRESSION_KIND(node)==comp && AST_TYPE(EXPRESSION_OP1(node))==string_type) {
318  // op2 is also a string (analysis has verified that point)
319  // â¹ We are comparing strings, generate a strcmp
320  emit("(strcmp("); code(EXPRESSION_OP1(node));
321  emit(", "); code(EXPRESSION_OP2(node));
322  emit(") %s 0)", name);
323  } else if (strcmp(name, "**") == 0) {
324  // power operator
325  emit("_toy_powint("); code(EXPRESSION_OP1(node));
326  emit(", "); code(EXPRESSION_OP2(node));
327  emit(")");
328  } else {
329  // simple binary operators
330  code(EXPRESSION_OP1(node));
331  emit(" %s ", name);
332  code_expr_cast(EXPRESSION_OP2(node));
333  }
334  break;
335  case 3: /* the only ternary operator is expr ? expr : expr */
336  code(EXPRESSION_OP1(node)); emit (" ? "); code(EXPRESSION_OP2(node));
337  emit(" : "); code(EXPRESSION_OP3(node));
338  break;
339  }
340 }

347 void produce_code_function(ast_node *node) {
348  struct s_function *n = (struct s_function *) node;
349  char *class_name = (n->in_class) ? IDENT_VAL(CLASS_NAME(n->in_class)): NULL;
350  int comma = 0;
351 
352  // Emit function type and name
353  code(AST_TYPE(node)); emit(" "); emit_function_name(node);
354 
355  // Produce parameters
356  emit("(");
357  if (!class_name && list_size(n->parameters) == 0)
358  emit("void");
359  else {
360  // Eventually add the special this parameter
361  if (class_name) {
362  emit("%s this", class_name);
363  comma += 1;
364  }
365  // produce "normal" parameters
366  FORLIST(p, n->parameters) {
367  ast_node *v = list_item_data(p);
368  if (comma++) emit(", ");
369  code(AST_TYPE(v)); emit(" "); code(v);
370  }
371  }
372  emit(")");
373 
374  // Body or ';'
375  if (n->body) { emit(" "); code(n->body); } else emit (";\n");
376 }

379 void produce_code_call(ast_node *node) {
380  struct s_call *n = (struct s_call *) node;
381  int comma = 0;
382 
383  if (IDENT_PREFIX(n->callee) == NULL) { // ---- Function call
384  code(n->callee); emit("(");
385  }
386  else { // ---- Method call
387  ast_node *method = IDENT_QUALIFIED(n->callee);
388 
389  emit("_TOY_INVOKE("); code(AST_TYPE(method)); emit(", "); // Result type
390  code(IDENT_PREFIX(n->callee)); // this
391  emit(", %d", FUNCTION_INDEX(method)); // method index in vtable
392  emit(", _this"); // First method parameter
393  comma++; // if method has params
394  }
395 
396  // produce the parameters
397  FORLIST(p, n->parameters) {
398  if (comma++) emit(", ");
399  code_expr_cast(list_item_data(p));
400  }
401  emit(")");
402 }
